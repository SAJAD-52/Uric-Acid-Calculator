import tkinter as tk
from tkinter import *
from tkinter.messagebox import *
from tkinter import messagebox


def show_answer():
    blank.delete(first=0, last=22)
    Femme_acd = float(num1.get())
    if 30 <= Femme_acd <= 50:
        messagebox.showinfo("Normal", "L'acide urique est normal, maintenez votre régime!")
    elif Femme_acd < 30:
        messagebox.showinfo("Attention",
                            "L'acide urique est bas!\nil s'agit d'une hypo-uricémie, veuillez manger les aliments "
                            "riches en purine comme la viande fumées, l'anchois et les crustacées!")
    else:
        messagebox.showinfo("Attention",
                            "L'acide urique est hors norme:\nIl s'agit d'une hyper-urécimie; veuillez manger des "
                            "aliments pauvres en purineet consulter votre médecin")

main = Tk()

main.config(bg="blue")
